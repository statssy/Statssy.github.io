## [백준 문제] P10819 차이를 최대로(파이썬)


내 답안
```python
# 1. 다음 순열을 만드는 함수
def next_permutation(a):
    n = len(a) - 1
    i = n
    while i > 0 and a[i-1] >= a[i]:
        i -= 1
    if i == 0: # 마지막 순열
        return False
    j = n
    while a[i-1] >= a[j]: # 오른쪽에 있으면서 a[i-1]보다 큰 수
        j -= 1
    a[i-1], a[j] = a[j], a[i-1] # SWAP
    j = n
    while i < j:
        a[i], a[j] = a[j], a[i] # a[i]부터 순열 뒤집기
        i += 1
        j -= 1
    return True

# 2. 공식 계산
def calculate(a) :
    ans = 0
    for i in range(len(a)-1) :        
        plus = abs(a[i]-a[i+1])
        ans = ans + plus
    
    return ans

# 3. 순열 하나씩 계산
n = int(input())
a = list(map(int, input().split()))
a = sorted(a)
ans = 0
while True :
    if next_permutation(a) is False :
        break
    temp = calculate(a);
    ans = max(ans, temp);

print(ans)
```
