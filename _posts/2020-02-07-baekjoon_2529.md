## [백준 문제] 2529번 부등호 (파이썬)

### 내 풀이 (시간 초과)
```python
import itertools
k = int(input())
ineq = list(map(str, input().split()))
all_lst = [str(i) for i in range(0,10)]
perm = list(itertools.permutations(all_lst, k+1))

for x in perm :
    x = list(x)
    is_possi = True
    for i in range(k) :
        if ineq[i] == '>' :
            if int(x[i]) > int(x[i+1]) :
                is_possi = is_possi & True
            else :
                is_possi = is_possi & False
        else :
            if int(x[i]) < int(x[i+1]) :
                is_possi = is_possi & True
            else :
                is_possi = is_possi & False
    if is_possi is True :
        ans1 = x
        break

for x in reversed(perm) :
    x = list(x)
    is_possi = True
    for i in range(k) :
        if ineq[i] == '>' :
            if int(x[i]) > int(x[i+1]) :
                is_possi = is_possi & True
            else :
                is_possi = is_possi & False
        else :
            if int(x[i]) < int(x[i+1]) :
                is_possi = is_possi & True
            else :
                is_possi = is_possi & False
    if is_possi is True :
        ans2 = x
        break


for i in range(k + 1):
    print(ans1[i], end='')
print()
for i in range(k + 1):
    print(ans2[i], end='')
```
